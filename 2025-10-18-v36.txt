    goimports -w .

    go mod tidy

    golangci-lint run

    go test $(PKG) -race -count=1 -coverprofile=$(COVER)

    @go tool cover -func=$(COVER) | tail -n1

    @TOTAL=$(go tool cover -func=$(COVER) | grep total | awk '{print $3}' | sed 's/%//'); \
